使用代理

## 获取主机 IP
## 主机 IP 保存在 /etc/resolv.conf 中
export hostip=$(cat /etc/resolv.conf |grep -oP '(?<=nameserver\ ).*')

## 配置代理
export http_proxy="socks5://${hostip}:7890"
export https_proxy="socks5://${hostip}:7890"
## 或者
export all_proxy="socks5://${hostip}:7890"

## 取消配置
unset all_proxy
